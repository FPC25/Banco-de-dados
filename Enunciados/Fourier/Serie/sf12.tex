\begin{itemize}
    \item[\textbf{a)}] Determine $x(t)$, solução geral da equação diferencial acima, para $\lambda = 0$;
    \item[\textbf{b)}] Determine $x(t)$, solução geral da equação diferencial acima, para $\lambda = 5$.
\end{itemize}
