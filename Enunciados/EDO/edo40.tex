Dada a E.D.O.:
\begin{equation*}
    y''(t) + 4y'(t) + 3y(t) = e^{-t}cos(t)
\end{equation*}

pede-se:
\begin{itemize}
    \item[\textbf{a)}] Determine a solução geral dessa E.D.O.;
    \item[\textbf{b)}] Determine uma solução particular que atenda as seguintes condições iniciais: $y(0) = 0$ e $y'(0) = 1$.
\end{itemize}
